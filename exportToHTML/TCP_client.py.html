32       def transfer(self, path):
33           if os.path.exists(path):
34               f = open(path, 'rb')
35               packet = f.read(1024)
36               while len(packet) > 0:
37                   self.conn.send(packet)
38                   packet = f.read(1024)
39               self.conn.send('DONE'.encode('ISO-8859-1', errors='ignore'))
40           else:
41               self.conn.send('File not found'.encode('ISO-8859-1', errors='ignore'))
42   
43       def download(self, command):
44           f = open(command, 'wb')
45           while True:
46               bits = self.conn.recv(1024)
47               if bits.endswith('DONE'.encode('ISO-8859-1', errors='ignore')):
48                   f.write(bits[:-4])
49                   f.close()
50                   print('[+] Transfer completed ')
51                   break
52               if 'File not found'.encode('ISO-8859-1', errors='ignore') in bits:
53                   print('[-] Unable to find out the file')
54                   break
55               f.write(bits)
