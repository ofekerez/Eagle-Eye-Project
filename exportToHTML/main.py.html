295  def map_network():
296      import threading
297      subnet_mask = request.form.get("subnet")
298      clients = []
299      threads = []
300      LOCK = threading.Lock()
301      count = 0
302      lists = [[] for i in range(helper_methods.get_processor_num() * 2)]
303      if subnet_mask:
304          result = helper_methods.check_hosts(subnet_mask)
305          for address in result.split('\n')[:-1]:
306              lists[count].append(address)
307              if count == helper_methods.get_processor_num() * 2 - 1:
308                  count = 0
309              else:
310                  count += 1
311          print(lists)
312          for i in range(len(lists)):
313              t = threading.Thread(target=helper_methods.scanner, args=(lists[i], LOCK, clients))
314              threads.append(t)
315              t.start()
316  
317          for thread in threads:
318              thread.join()
319          return render_template("ActiveIPs.html", content=clients)
320      return render_template("ActiveIPs.html")
