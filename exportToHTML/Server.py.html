22       def run(self) -> None:
23           while True:
24               length = self.conn.recv(1024).decode()
25               while not length:
26                   length = self.conn.recv(1024).decode()
27               msg = self.conn.recv(int(length)).decode()
28               print(msg)
29               if msg == 'SNF_SRT':
30                   st = ''
31                   print('Sniffing Started')
32                   sorted_packets, path = snf.gen_sniff()
33                   st += snf.filter_HTTP(sorted_packets[0]) + snf.filter_ICMP(sorted_packets[1]) + snf.filter_SMB(
34                       sorted_packets[2])
35                   st += snf.filter_FTP(sorted_packets[3]) + snf.filter_SSH(sorted_packets[4]) + snf.filterstringDNS(
36                       sorted_packets[5]) + snf.filter_DHCP(sorted_packets[6])
37                   self.conn.send(str(len(st)).encode())
38                   self.conn.send(st.encode('ISO-8859-1', errors='ignore'))
39                   time.sleep(3)
40                   self.transfer(path)
41                   continue
42               elif msg == 'SYN_SRT':
43                   open_ports = PortScanner(get_ip_address()).SYN_Scan_Wrap()
44                   st = ''
45                   for open_port in open_ports:
46                       st += f"Port {open_port} is open!" + '\n'
47                   self.conn.send(str(len(st)).encode('ISO-8859-1', errors='ignore'))
48                   self.conn.send(st.encode('ISO-8859-1', errors='ignore'))
49                   continue
50               elif msg == 'STEALTH_SRT':
51                   open_ports = PortScanner(get_ip_address()).Stealth_Scan_Wrap()
52                   st = ''
53                   for open_port in open_ports:
54                       st += f"Port {open_port} is open!" + '\n'
55                   self.conn.send(str(len(st)).encode('ISO-8859-1', errors='ignore'))
56                   self.conn.send(st.encode('ISO-8859-1', errors='ignore'))
57                   continue
58               elif msg == 'UDP_SRT':
59                   open_ports = PortScanner(get_ip_address()).UDP_Scan_Wrap()
60                   st = ''
61                   for open_port in open_ports:
62                       st += f"Port {open_port} is open!" + '\n'
63                   self.conn.send(str(len(st)).encode('ISO-8859-1', errors='ignore'))
64                   self.conn.send(st.encode('ISO-8859-1', errors='ignore'))
65                   continue
66               elif msg == 'REV_ACT':
67                   Client(self.addr[0], 9999).run()
68               elif msg == 'EXIT':
69                   self.__init__()

71       def transfer(self, path):
72           import os
73           if os.path.exists(path):
74               f = open(path, 'rb')
75               packet = f.read(1024)
76               while len(packet) > 0:
77                   self.conn.send(packet)
78                   packet = f.read(1024)
79               self.conn.send('DONE'.encode('ISO-8859-1', errors='ignore'))
80           else:
81               self.conn.send('File not found'.encode('ISO-8859-1', errors='ignore'))
