5    def check_ports(start_port, end_port):
6        if start_port > end_port:
7            start_port, end_port = end_port, start_port
8        elif start_port == end_port:
9            end_port += 1
10       return start_port, end_port

13   def Connect_Scan(IP_address, start_port=1, end_port=65536):
14       """TCP S flag stands for SYN request in the TCP 3 way handshake. 
15       TCP A flag stands for ACK response in the TCP 3 way handshake 
16       The code for SYN - ACK flag is 0x12."""
17       open_ports = []
18       start_port, end_port = check_ports(start_port, end_port)
19       for port in range(start_port, end_port):
20           packet = IP(dst=IP_address) / TCP(dport=port, flags='S')
21           response = sr1(packet, timeout=0.5, verbose=0)
22           if response and response.haslayer(TCP) and response.getlayer(TCP).flags == 0x12:
23               print(f"Port {port} is open!")
24               open_ports.append(port)
25               ACK = IP(dst=IP_address) / TCP(dport=response.sport, flags='AR')
26               sr(ACK, timeout=0.2, verbose=0)
27       print("Scan is complete!")
28       return open_ports

57   def UDP_Scan(dst_ip, start_port=1, end_port=65535):
58       start_port, end_port = check_ports(start_port, end_port)
59       open_ports = []
60       for port in range(start_port, end_port):
61           response = sr1(IP(dst=dst_ip) / UDP(dport=port), timeout=10, verbose=0)
62           if not response:
63               print(f"Port {port} is Filtered or Open!")
64           elif response.haslayer(UDP):
65               open_ports.append(port)
66               print(f"Port {port} is Open!")
67           elif response.haslayer(ICMP) and int(response.getlayer(ICMP).type) == 3 and int(
68                   response.getlayer(ICMP).code) in [1, 2, 9, 10, 13]:
69               print(f"Port {port} is Filtered!")
70           else:
71               print(f"Port {port} is Closed!")
72       return open_ports
