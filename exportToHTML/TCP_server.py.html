102      def on_press(self, key):
103          if key == Key.up:
104              for i in range(len(self.current_input) + 10):
105                  self.controller.press(Key.backspace)
106                  self.controller.release(Key.backspace)
107              self.current_input = ''
108              for char in self.Back():
109                  self.controller.press(char)
110                  self.controller.release(char)
111                  self.current_input += char
112          elif key == Key.down:
113              for i in range(len(self.current_input) + 10):
114                  self.controller.press(Key.backspace)
115                  self.controller.release(Key.backspace)
116              self.current_input = ''
117              for char in self.Forward():
118                  self.controller.press(char)
119                  self.controller.release(char)
120                  self.current_input += char
121          elif key == Key.backspace:
122              self.current_input = self.current_input[:-1]
123          elif key == Key.enter:
124              self.current_input = ''
125          else:
126              try:
127                  self.current_input += key.char
128              except Exception:
129                  pass

131      def Back(self):
132          """Up arrow has been pressed"""
133          # self.commands: ['dir', 'ipconfig', 'cd ..']
134          ind = self.ind
135          try:
136              self.ind += 1
137              if self.ind > len(self.commands):
138                  self.ind = 0
139              return self.commands[ind]
140          except IndexError:
141              return ''

143      def SaveObject(self, obj: ...):
144          """ 
145          Appends a command to the list 
146          """
147          if obj in self.commands:  # If I pressed a command which I already executed bring it to the first place.
148              self.commands.remove(obj)
149              self.commands.insert(0, obj)
150          else:
151              self.commands.insert(0, obj)
152              self.ind = self.commands.index(obj)
153  
154      def Forward(self) -> str:
155          """ 
156          Down arrow has been pressed. 
157          """
158          try:
159              self.ind -= 1
160              ind = self.ind
161              return self.commands[ind]
162          except IndexError:
163              return ''
